#Chargement et pr-traitement des donnes
hlth_data = pd.read_csv('hlth_cd_anr_1_Data.csv')
hlth_data['Value'] = hlth_data['Value'].apply(lambda x : x.replace(',','').replace(':','-1'))
hlth_data['Value'] = hlth_data['Value'].astype('int')
hlth_data1 = hlth_data
hlth_data1 = hlth_data1[hlth_data['AGE'] =='Total']
hlth_data1 = hlth_data1[hlth_data['SEX']=='Total']

#Calcul du nombre total de cas (sur toutes les anns) par pays et par maladie
hlth_data_grouped = hlth_data1.groupby(['GEO','ICD10'],as_index=False)['Value'].sum()

#On charge les codes de chaque pays (ncessaire pour la vizualisation de la map d'europe sur altair)
#Pas trouv d'autre moyen pour le faire
countries_codes = pd.read_excel('countries_codes.xlsx')

#rapide prtraitement du nom des pays pour les adapters aux noms de notre dataset (des maladies)
countries_codes['Country name'] = countries_codes['Country name'].apply(lambda x : 'United Kingdom' if x =='United Kingdom of Great Britain and Northern Ireland' else x)
countries_codes['Country name'] = countries_codes['Country name'].apply(lambda x : 'France (metropolitan)' if x =='France' else x)
countries_codes['Country name'] = countries_codes['Country name'].apply(lambda x : 'Germany (until 1990 former territory of the FRG)' if x =='Germany' else x)

#On effectue un merge pour ajouter l'information des codes de chaque pays
hlth_map_data = pd.merge(hlth_data_grouped,countries_codes, left_on='GEO', right_on='Country name',how='left')
hlth_map_data_filtered = hlth_map_data.dropna()

#On effectue un pivot sur les maladies (ncessaire pour pouvoir selectioner les maladies
#dans notre visualisation)
test1 = hlth_map_data_filtered.pivot(index = ['GEO'],columns=['ICD10'],values=['Value'])
test1 = test1['Value'].reset_index()

#Remerge car l'info sur les codes pays a t perdue  cause du pivot
test2 = pd.merge(test1,countries_codes, left_on='GEO', right_on='Country name',how='left')

hlth_data4 = hlth_data3.copy()

#Modification colonne AGE

hlth_data4['AGE'] = hlth_data4['AGE'].str.replace("85 years or over","From 85 to over")
hlth_data4['AGE'] = hlth_data4['AGE'].str.replace('From 1 to 4 years','From 01 to 04 years')
hlth_data4['AGE'] = hlth_data4['AGE'].str.replace('From 5 to 9 years','From 05 to 09 years')
hlth_data4 = hlth_data4.iloc[:-2,:]

#Fusion des lignes (2 + 38) et (3 + 39) pour avoir les ages de 0  4 ans pour les femmmes et les hommmes
#Females
dicF_1 = {'AGE':'From 0 to 04 years', 'SEX':'Females'}
dicF_2 = {cols:(hlth_data4.loc[2,cols] + hlth_data4.loc[38,cols]) for cols in hlth_data4.columns[2:]}
dicF = {**dicF_1,**dicF_2}

#Males
dicM_1 = {'AGE':'From 0 to 04 years', 'SEX':'Males'}
dicM_2 = {cols:(hlth_data4.loc[3,cols] + hlth_data4.loc[39,cols]) for cols in hlth_data4.columns[2:]}
dicM = {**dicM_1,**dicM_2}

hlth_data4 = hlth_data4.append(dicF, ignore_index=True)
hlth_data4 = hlth_data4.append(dicM, ignore_index=True)

#Suppression ligne 2,3,38,39

hlth_data4.drop( hlth_data4[hlth_data4['AGE'] == "Less than 1 year"].index, inplace=True)
